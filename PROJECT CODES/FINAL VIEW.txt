CREATE VIEW DIAGSUM AS
SELECT CLINIC.FIRST_NAME, CLINIC.LAST_NAME, DIAGNOSIS.D_NO, DIAGNOSISHISTORY.D_DATE
FROM CLINIC
INNER JOIN DIAGNOSIS ON CLINIC.PERSON_ID = DIAGNOSIS.PERSON_ID
INNER JOIN DIAGNOSISHISTORY ON DIAGNOSIS.D_NO = DIAGNOSISHISTORY.D_NO

CREATE VIEW PESUM AS
SELECT CLINIC.FIRST_NAME, CLINIC.LAST_NAME, PHYSICALEXAM.PE_NO, PHYSICALEXAMHISTORY.PE_DATE
FROM CLINIC
INNER JOIN PHYSICALEXAM ON CLINIC.PERSON_ID = PHYSICALEXAM.PERSON_ID
INNER JOIN PHYSICALEXAMHISTORY ON PHYSICALEXAM.PE_NO = PHYSICALEXAMHISTORY.PE_NO

CREATE VIEW PCSUM AS
SELECT CLINIC.FIRST_NAME, CLINIC.LAST_NAME, PERSONALCONSULTATION.PC_NO, PERSONALCONSULTATIONHISTORY.PC_DATE
FROM CLINIC
INNER JOIN PERSONALCONSULTATION ON CLINIC.PERSON_ID = PERSONALCONSULTATION.PERSON_ID
INNER JOIN PERSONALCONSULTATIONHISTORY ON PERSONALCONSULTATION.PC_NO = PERSONALCONSULTATIONHISTORY.PC_NO
